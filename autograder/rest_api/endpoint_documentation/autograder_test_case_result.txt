/autograder_test_case_results/<id>/
GET: Returns the specified test case result.

Access rules: Same as GET /submissions/<submission id>/

Response format:
{
    "type": "autograder_test_case_result",
    "id": <id>,

    "test_case_name": <autograder_test_case>,

    "urls": {
        "self": "/autograder_test_case_results/<id>/",
        "submission": "/submissions/<id>/"
    },

    //** NOTE: Some or all of the following may be omitted **//
    //** depending on the feedback level and the way the    **//
    //** test case determines correctness.                  **//

    "return_code_correct": <bool>,
    "expected_return_code": <int>,
    "actual_return_code": <int>,

    "output_correct": <bool>,

    // At most one of standard_output_diff or standard_output will be present
    // depending on the feedback level and how the test case
    // evaluates correctness.
    "standard_output": <output>,
    "standard_output_diff": <diff>,
    
    // Similarly for standard_error_output_diff and standard_error_output
    "standard_error_output": <output>,
    "standard_error_output_diff": <diff>,

    "valgrind_errors_present": <bool>,
    "valgrind_output": <output>,

    "compilation_succeeded": <bool>,
    "compilation_stdout": <string>,
    "compilation_stderr": <string>,

    "return_code_points_awarded": <int>,
    "return_code_points_possible": <int>,

    "output_points_awarded": <int>,
    "output_points_possible": <int>,

    "valgrind_points_awarded": <int>,
    "valgrind_points_possible": <int>,

    "compilation_points_awarded": <int>,
    "compilation_points_possible": <int>,

    "total_points_awarded": <total int>,
    "total_points_possible": <int>,

    "timed_out": <bool>
}
